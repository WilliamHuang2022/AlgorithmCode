def overlap(a,b):
    if not a and b:
        return b
    elif not b and a:
        return a
    elif not b and not a:
        return []
    # 保证a在b左边
    if a[0]>b[0]:
        tmp=b
        b=a
        a=tmp
    # 相交
    if a[0]<b[0] and a[1]>=b[0] and a[1]<b[1]:
        return [b[0],a[1]]
    # 相离
    if a[1]<b[0]:
        return []
    # 包含
    if a[0]<=b[0] and b[1]<=a[1]:
        return b
    if a[0]<=b[0] and a[1]<=b[1]:
        return a
